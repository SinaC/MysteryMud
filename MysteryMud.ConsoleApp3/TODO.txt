        public void RemoveEffectGroup(World world, int groupId) // could be used to dispel effects from a spell or item
        {
            var query = new QueryDescription()
                .WithAll<EffectGroup>();
        
            world.Query(in query, (Entity e, ref EffectGroup group) =>
            {
                if (group.GroupId == groupId)
                {
                    e.Destroy();
                }
            });
        }
